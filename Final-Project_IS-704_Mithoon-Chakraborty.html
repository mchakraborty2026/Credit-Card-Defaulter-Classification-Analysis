# Rename target variable for easier use
credit <- credit %>%
  rename(Default = default.payment.next.month)

# Remove ID because it is only an identifier, not a meaningful predictor
credit <- credit %>%
  select(-ID)

# Convert target to factor for classification
credit$Default <- factor(credit$Default,
                         levels = c(0, 1),
                         labels = c("No_Default", "Default"))

# Clean categorical predictors

credit <- credit %>%
  mutate(
    SEX = factor(SEX,
                 levels = c(1, 2),
                 labels = c("Male", "Female")),
    
    EDUCATION = case_when(
      EDUCATION == 1 ~ "Graduate_School",
      EDUCATION == 2 ~ "University",
      EDUCATION == 3 ~ "High_School",
      EDUCATION %in% c(0, 4, 5, 6) ~ "Other",
      TRUE ~ "Other"
    ),
    
    MARRIAGE = case_when(
      MARRIAGE == 1 ~ "Married",
      MARRIAGE == 2 ~ "Single",
      MARRIAGE == 3 ~ "Other",
      MARRIAGE == 0 ~ "Other",
      TRUE ~ "Other"
    ),
    
    EDUCATION = factor(EDUCATION),
    MARRIAGE = factor(MARRIAGE)
  )
